import { NgModule } from '@angular/core'
import { ReactiveFormsModule } from '@angular/forms'
import { SharedModule } from '../../shared/shared.module'

        FSelectComponent,
        FTextAreaComponent,
        FPasswordComponent,
        FDateComponent,
        FCheckboxComponent,
        FRadiobuttonComponent,
        FErrorComponent,
        FExplanationComponent,
    ],
})
export class FormElementsModule {}
